156  Real2DVector GraphLayout::getNormalizedDistanceVector(nid_t n1, nid_t n2)
157  {
158  const float x1 = getX(n1);
159  const float x2 = getX(n2);
160  const float y1 = getY(n1);
161  const float y2 = getY(n2);
162  const float dx = x2 - x1;
163  const float dy = y2 - y1;
164  const float len = std::sqrt(dx*dx + dy*dy);
165 
166  return Real2DVector(dx / len, dy / len);

222  void GraphLayout::writeToPNG(const int image_w, const int image_h,
223  std::string path)
224  {
225  const float xRange = getXRange();
226  const float yRange = getYRange();
227  const RPGraph::Coordinate center = getCenter(); // Why do we need the center?
228  const float xCenter = center.x; // Probably to set the origin for layout.
229  const float yCenter = center.y;
230  const float minX = xCenter - xRange/2.0;
231  const float minY = yCenter - yRange/2.0;
232  const float xScale = image_w/xRange;
233  const float yScale = image_h/yRange;
234 
235  // Here we need to do some guessing as to what the optimal
236  // opacity of nodes and edges might be, given network size.
240  const float node_opacity = 10000.0 / graph.num_nodes();
241  const float edge_opacity = 100000.0 / graph.num_edges();
242 
243  // Write to file.
244  pngwriter layout_png(image_w, image_h, 0, path.c_str());
245  layout_png.invert(); // set bg. to white
246 
247  for (nid_t n1 = 0; n1 < graph.num_nodes(); ++n1)
248  {
249  // Plot node,
250  layout_png.filledcircle_blend((getX(n1) - minX)*xScale,
251  (getY(n1) - minY)*yScale,
252  3, node_opacity, 0, 0, 0);
253  for (nid_t n2 : graph.neighbors_with_geq_id(n1)) {
254  // ... and edge.
255  layout_png.line_blend((getX(n1) - minX)*xScale, (getY(n1) - minY)*yScale,
256  (getX(n2) - minX)*xScale, (getY(n2) - minY)*yScale,
257  edge_opacity, 0, 0, 0);
258  }
259  }
260  // Write it to disk.
261  layout_png.write_png();
262  }
